import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from statsmodels.tsa.seasonal import seasonal_decompose
from statsmodels.tsa.holtwinters import ExponentialSmoothing

np.random.seed(42)
dates = pd.date_range(start='2010-01-01', periods=120, freq='M')
data = pd.Series(
    np.sin(np.linspace(0, 20, 120)) * 100 + np.random.normal(size=120) * 10,
    index=dates
)
data = pd.DataFrame(data, columns=['Passengers'])

plt.figure(figsize=(10, 6))
data.plot()
plt.title('Synthetic Monthly Number of Passengers')
plt.xlabel('Date')
plt.ylabel('Number of Passengers')
plt.show()

import pandas as pd
from sklearn.feature_extraction.text import CountVectorizer
from sklearn.naive_bayes import MultinomialNB
from sklearn.pipeline import make_pipeline
from sklearn.model_selection import train_test_split
from sklearn import metrics
import numpy as np

np.random.seed(42)

sentiments = ['positive', 'negative']
texts = ['I love this movie!' if np.random.rand() > 0.5 else 'I hate this movie!' for _ in range(1000)]
labels = [np.random.choice(sentiments) for _ in range(1000)]

df = pd.DataFrame({'text': texts, 'sentiment': labels})

X = df['text']
y = df['sentiment']

X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42)

model = make_pipeline(CountVectorizer(), MultinomialNB())
model.fit(X_train, y_train)

predictions = model.predict(X_test)

accuracy = metrics.accuracy_score(y_test, predictions)
print(f'Accuracy: {accuracy:.2f}')

sample_review = "I absolutely loved this movie. The plot was amazing!"
print(f'Predicted sentiment: {model.predict([sample_review])[0]}')

import pandas as pd
from sklearn.datasets import make_blobs
from sklearn.cluster import KMeans
from sklearn.decomposition import PCA
from sklearn.preprocessing import StandardScaler
import matplotlib.pyplot as plt

X, y = make_blobs(n_samples=300, centers=3, cluster_std=1.0, random_state=42)
df = pd.DataFrame(X, columns=['Feature1', 'Feature2'])
df['species'] = y

scaler = StandardScaler()
scaled_features = scaler.fit_transform(df[['Feature1', 'Feature2']])

kmeans = KMeans(n_clusters=3, random_state=42)
clusters = kmeans.fit_predict(scaled_features)

df['cluster'] = clusters

pca = PCA(n_components=2)
pca_result = pca.fit_transform(scaled_features)
df_pca = pd.DataFrame(pca_result, columns=['PC1', 'PC2'])
df_pca['cluster'] = clusters

plt.figure(figsize=(10, 6))
plt.scatter(df_pca['PC1'], df_pca['PC2'], c=df_pca['cluster'], cmap='viridis')
plt.title('Synthetic K-Means Clustering (PCA Reduced)')
plt.xlabel('Principal Component 1')
plt.ylabel('Principal Component 2')
plt.colorbar(label='Cluster')
plt.show()
